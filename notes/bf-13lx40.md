pub fn exit_to_scope(&mut self, target_indent: usize) {
    // 1. Guard against invalid exits (to deeper levels)
    if target_indent > self.current_indent() {
        return;
    }

    // 2. Prevent emptying the stack (always keep root)
    if would_be_empty {
        retain only root;
        return;
    }

    // 3. Remove all scopes deeper than target
    retain(|s| s.indent_level <= target_indent);

    // 4. Find closest parent if exact target missing
    if !scope_at_target_exists() {
        find_closest_parent_scope();
    }
}
