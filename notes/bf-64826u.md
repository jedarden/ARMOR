from tests import validate_standard_error_response, validate_error_response

# Basic usage
response = {"error": "not_found", "message": "Resource not found"}
validate_standard_error_response(response)

# With custom spec
from tests import ErrorResponseSpec
custom_spec = ErrorResponseSpec(
    required_fields={'error', 'message', 'code'},
    optional_fields={'details', 'request_id'}
)
validate_error_response(response, spec=custom_spec)

# With custom validators
def validate_code(value, field_name):
    if not isinstance(value, int):
        return f"{field_name} must be an integer"
    if value < 400 or value >= 600:
        return f"{field_name} must be a 4xx or 5xx code"
    return None

validate_error_response(
    response,
    custom_validators={"code": validate_code}
)
